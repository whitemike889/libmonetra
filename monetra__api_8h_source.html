17 /*! \defgroup LM_api Monetra API
18  * Monetra API based on mstdlib. The supported next-generation library.
19  * Use \#include <monetra_api.h>.
20  *
21  * Simple usage example below:
22  *
23  * \code{.c}
24  * #include <monetra_api.h>
25  *
26  * static void trans_callback(LM_conn_t *conn, M_event_t *event, LM_event_type_t type, LM_trans_t *trans)
27  * {
28  * M_list_str_t *params;
29  * size_t i;
30  *
31  * switch (type) {
32  * case LM_EVENT_CONN_CONNECTED:
33  * M_printf("Connected successfully\n");
34  * break;
35  *
36  * case LM_EVENT_CONN_DISCONNECT:
37  * case LM_EVENT_CONN_ERROR:
38  * M_printf("Disconnect or Error received: %s\n", LM_conn_error(conn));
39  * LM_conn_destroy(conn);
40  * M_event_done(event);
41  * break;
42  *
43  * case LM_EVENT_TRANS_DONE:
44  * M_printf("Transaction %p complete:\n", trans);
45  * params = LM_trans_response_keys(trans);
46  * for (i=0; i<M_list_str_len(params); i++) {
47  * const char *key = M_list_str_at(params, i);
48  * M_printf("\t'%s' = '%s'\n", key, LM_trans_response_param(trans, key));
49  * }
50  * M_list_str_destroy(params);
51  * LM_trans_delete(trans);
52  * // We're only running a single transaction, issue a graceful disconnect
53  * LM_conn_disconnect(conn);
54  * break;
55  *
56  * case LM_EVENT_TRANS_ERROR:
57  * case LM_EVENT_TRANS_NOCONNECT:
58  * M_printf("Transaction %p error (connectivity): %s\n", trans, LM_conn_error(conn));
59  * LM_trans_delete(trans);
60  * // We should still get an LM_EVENT_CONN_ERROR after this event for additional cleanup.
61  * break;
62  * }
63  * }
64  *
65  * int main(int argc, char **argv)
66  * {
67  * M_event_t *event = M_event_create(M_EVENT_FLAG_NONE);
68  * LM_conn_t *conn = LM_conn_init(event, trans_callback, "testbox.monetra.com", 8665);
69  * LM_trans_t *trans;
70  *
71  * LM_conn_connect(conn);
72  *
73  * trans = LM_trans_new(conn);
74  * LM_trans_set_param(trans, "username", "test_retail:public");
75  * LM_trans_set_param(trans, "password", "publ1ct3st");
76  * LM_trans_set_param(trans, "action", "sale");
77  * LM_trans_set_param(trans, "account", "4012888888881881");
78  * LM_trans_set_param(trans, "expdate", "1220");
79  * LM_trans_set_param(trans, "zip", "32606");
80  * LM_trans_set_param(trans, "cv", "999");
81  * LM_trans_set_param(trans, "amount", "12.00");
82  * LM_trans_send(trans);
83  *
84  * M_event_loop(event, M_TIMEOUT_INF);
85  *
86  * // NOTE: we cleaned up 'conn' within the trans_callback. We can't have
87  * // exited the event loop otherwise.
88  * conn = NULL;
89  *
90  * M_event_destroy(event);
91  *
92  * M_library_cleanup();
93  *
94  * return 0;
95  * }
96  * \endcode
